	\begin{lstlisting}
	void run_mul(mulfunc f, const char* name,
             int sz, double a[sz][sz], double b[sz][sz],
             double c[sz][sz])
{
	   double start, stop, mflops;

	   SSIM_FLUSH_CACHE;
	   
	   start = gettime();
	   
	   init(sz, c, 0);
	   (*f)(sz, a, b, c);
	   
	   stop = gettime();
	   mflops = 2.0*sz*sz*sz / (stop-start) / 1000000.0;
	   
	   fprintf(stderr, "  %s: %f sec. : %f MFlops\n", 
	           name, stop - start, mflops);
	   printf("%f ", mflops);
}
	\end{lstlisting}
